00036 /* Synthesize Singleton For Class
00037  *
00038  * Creates a singleton interface for the specified class with the following methods:
00039  *
00040  * + (MyClass*) sharedInstance;
00041  * + (void) purgeSharedInstance;
00042  *
00043  * Calling sharedInstance will instantiate the class and swizzle some methods to ensure
00044  * that only a single instance ever exists.
00045  * Calling purgeSharedInstance will destroy the shared instance and return the swizzled
00046  * methods to their former selves.
00047  *
00048  * 
00049  * Usage:
00050  *
00051  * MyClass.h:
00052  * ========================================
00053  *  #import "SynthesizeSingleton.h"
00054  *
00055  *  @interface MyClass: SomeSuperclass
00056  *  {
00057  *      ...
00058  *  }
00059  *  SYNTHESIZE_SINGLETON_FOR_CLASS_HEADER(MyClass);
00060  *
00061  *  @end
00062  * ========================================
00063  *
00064  *
00065  *  MyClass.m:
00066  * ========================================
00067  *  #import "MyClass.h"
00068  *
00069  *  @implementation MyClass
00070  *
00071  *  SYNTHESIZE_SINGLETON_FOR_CLASS(MyClass);
00072  *
00073  *  ...
00074  *
00075  *  @end
00076  * ========================================
00077  *
00078  *
00079  * Note: Calling alloc manually will also initialize the singleton, so you
00080  * can call a more complex init routine to initialize the singleton like so:
00081  *
00082  * [[MyClass alloc] initWithParam:firstParam secondParam:secondParam];
00083  *
00084  * Just be sure to make such a call BEFORE you call "sharedInstance" in
00085  * your program.
00086  */

00221 /* A lesser singleton has a shared instance, but can also be instantiated on its own.
00222  *
00223  * For a lesser singleton, you still use SYNTHESIZE_SINGLETON_FOR_CLASS_HEADER(),
00224  * but use SYNTHESIZE_LESSER_SINGLETON_FOR_CLASS() in the implementation file.
00225  * You must specify which creation methods are to initialize the shared instance
00226  * (besides "sharedInstance") via CALL_LESSER_SINGLETON_INIT_METHOD()
00227  *
00228  * Example:
00229  *
00230  * MyClass.h:
00231  * ========================================
00232  *  #import "SynthesizeSingleton.h"
00233  *
00234  *  @interface MyClass: SomeSuperclass
00235  *  {
00236  *      int value;
00237  *      ...
00238  *  }
00239  *  SYNTHESIZE_SINGLETON_FOR_CLASS_HEADER(MyClass);
00240  *
00241  *  + (void) initSharedInstanceWithValue:(int) value;
00242  *
00243  * - (id) initWithValue:(int) value;
00244  *
00245  *  @end
00246  * ========================================
00247  *
00248  *
00249  *  MyClass.m:
00250  * ========================================
00251  *  #import "MyClass.h"
00252  *
00253  *  @implementation MyClass
00254  *
00255  *  SYNTHESIZE_LESSER_SINGLETON_FOR_CLASS(MyClass);
00256  *
00257  *  + (void) initSharedInstanceWithValue:(int) value
00258  *  {
00259  *      CALL_LESSER_SINGLETON_INIT_METHOD(MyClass, initWithValue:value);
00260  *  }
00261  *
00262  *  ...
00263  *
00264  *  @end
00265  * ========================================
00266  *
00267  *
00268  * Note: CALL_LESSER_SINGLETON_INIT_METHOD() will not work if your
00269  * init call contains commas.  If you need commas (such as for varargs),
00270  * or other more complex initialization, use the PRE and POST macros:
00271  *
00272  *  + (void) initSharedInstanceComplex
00273  *  {
00274  *      CALL_LESSER_SINGLETON_INIT_METHOD_PRE(MyClass);
00275  *
00276  *      int firstNumber = [self getFirstNumberSomehow];
00277  *      _sharedInstance = [[self alloc] initWithValues:firstNumber, 2, 3, 4, -1];
00278  *
00279  *      CALL_LESSER_SINGLETON_INIT_METHOD_POST(MyClass);
00280  *  }
00281  */
